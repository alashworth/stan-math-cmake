240  cl::Context& ctx = opencl_context.context();
241  cl::CommandQueue& queue = opencl_context.queue();
242  buffer_cl_ = cl::Buffer(ctx, CL_MEM_READ_WRITE, sizeof(T) * size());
243  for (int i = 0, offset_size = 0; i < cols_; i++, offset_size += rows_) {
244  check_size_match("matrix constructor", "input rows", A[i].size(),
245  "matrix_cl rows", rows_);

250  sizeof(T) * offset_size, sizeof(T) * rows_, A[i].data(), nullptr,
